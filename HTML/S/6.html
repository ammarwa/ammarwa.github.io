  19 bool launchFinder(BasicBlock *block, void *launchCallPtr) {
  20   CallBase **ptr = (CallBase **)launchCallPtr;
  21 
  22   for (Instruction &inst : *block) {
  23 
  24     // If inst is a call inst look for cuda funktions
  25     if (CallBase *ci = dyn_cast_or_null<CallBase>(&inst)) {
  26       StringRef name = ci->getCalledFunction()->getName();
  27       if (name == "cudaSetupArgument")
  28         continue;
  29       if (name == "cudaLaunch") {
  30         *ptr = ci;
  31         return true;
  32         // Any other call could be the kernel wrapper
  33         // Check if cudaLaunch is called from the possible kernel wrapper
  34         // function
  35       } else {
  36 
  37         Function *cudaLaunch = block->getParent()->getParent()->getFunction("cudaLaunch");
  38         for (Value *launchCallVal : cudaLaunch->users()) {
  39           if (CallBase *launchCallBase = dyn_cast_or_null<CallBase>(launchCallVal)) {
  40             if (launchCallBase->getFunction() == ci->getCalledFunction()) {
  41               *ptr = launchCallBase;
  42               return true;
  43             }
  44           }
  45         } // end for (Value* launchCallVal : cudaLaunch->users())
  46 
  47       } // end else
  48     }   // end if (CallBase *ci = dync_cast_or_null<CallBase>(&inst))
  49     if (InvokeInst *invi = dyn_cast_or_null<InvokeInst>(&inst)) {
  50       // Check if cudaLaunch is called from the possible kernel wrapper function
  51       Function *cudaLaunch = block->getParent()->getParent()->getFunction("cudaLaunch");
  52       for (Value *launchCallVal : cudaLaunch->users()) {
  53         if (CallBase *launchCallBase = dyn_cast_or_null<CallBase>(launchCallVal)) {
  54           if (launchCallBase->getFunction() == invi->getCalledFunction()) {
  55             *ptr = launchCallBase;
  56             return true;
  57           }
  58         }
  59       } // end for (Value* launchCallVal : cudaLaunch->users())
  60 
  61     } // end if (CallBase *ci = dync_cast_or_null<CallBase>(&inst))
  62 
  63   } // end for (Insturction &inst : *block)
  64 
  65   return false;
  66 }

  76 void getKernelDescriptors(llvm::Module &m, std::vector<KernelDescriptor> &desc) {
  77   Function *registerFunction = m.getFunction("__cudaRegisterFunction");
  78   if (registerFunction == nullptr)
  79     return;
  80   for (auto *user : registerFunction->users()) {
  81     // check for callbase as this will find all call sites of "__cudaRegisterFunction"
  82     CallBase *callBase = dyn_cast_or_null<CallBase>(user);
  83     if (callBase != nullptr) {
  84       // 2nd argumen is the handle under which the kernel is registered
  85       Value *val_func = (callBase->getArgOperand(1)->stripPointerCastsAndAliases());
  86       Function *handle = dyn_cast_or_null<Function>(val_func);
  87       // 3rd argument is the kernel name (on device side)
  88       GlobalVariable *gv = dyn_cast_or_null<GlobalVariable>(callBase->getArgOperand(2)->stripPointerCastsAndAliases());
  89       ConstantDataSequential *val_name = nullptr;
  90       if (gv != nullptr)
  91         val_name = dyn_cast_or_null<ConstantDataSequential>(gv->getInitializer());
  92 
  93       if (val_func != nullptr && val_name != nullptr) {
  94         StringRef name = val_name->getAsString();
  95         // Remove null byte at end
  96         name = name.substr(0, name.size() - 1);
  97         desc.push_back({handle, name});
  98       }
  99     }
 100   }
 101   return;
 102 }

 104 // Finds the instructions that call the kernel launch wrapper function (klFun)
 105 // in this function cudaLaunch or cudaLaunchKernel will be called
 106 // this function will not return the call sites if the wrapper function is
 107 // already inlined
 108 void getKernelLaunchSites(llvm::Function *klFun, std::vector<llvm::CallBase *> &callSites) {
 109   for (auto *user : klFun->users()) {
 110     CallBase *callBase = dyn_cast_or_null<CallBase>(user);
 111     // user must be callbase and call the kernel launch function
 112     if (callBase != nullptr && (callBase->getCalledFunction() == klFun)) {
 113       callSites.push_back(callBase);
 114     }
 115   }
 116   return;
 117 }

 119 void getKernelArguments(llvm::CallBase *kernelLaunchSite, std::vector<llvm::Value *> &args) {
 120   for (Use &op : kernelLaunchSite->operands()) {
 121     // Make sure not to inclide the kernel launch wrapper
 122     Function *fun = dyn_cast_or_null<Function>(op.get());
 123     if (fun != nullptr and fun == kernelLaunchSite->getCalledFunction())
 124       continue;
 125     // also skip invoke operands which are basicblocks
 126     if (dyn_cast_or_null<BasicBlock>(op.get()) != nullptr)
 127       continue;
 128 
 129     args.push_back(op.get());
 130   }
 131   // get arguments regarding launch configuration
 132   return;
 133 }

 135 CallBase *getKernelConfigCall(llvm::Module &m, llvm::CallBase *kernelLaunchSite) {
 136   Function *confFunc = nullptr;
 137 
 138   if (usesNewKernelLaunch(m)) {
 139     confFunc = m.getFunction("__cudaPushCallConfiguration");
 140   } else {
 141     confFunc = m.getFunction("cudaConfigureCall");
 142   }
 143 
 144   assert(confFunc != nullptr);
 145 
 146   // BFSearch back to configuration call
 147   CallBase *confCall = nullptr;
 148   queue<BasicBlock *> q;
 149   q.push(kernelLaunchSite->getParent());
 150   while (q.empty() == false and confCall == nullptr) {
 151     BasicBlock *currentBlock = q.front();
 152     q.pop();
 153 
 154     if (currentBlock != kernelLaunchSite->getParent()) {
 155       // Search for configure call
 156       for (Instruction &inst : *currentBlock) {
 157         CallBase *ci = dyn_cast_or_null<CallBase>(&inst);
 158         if (ci != nullptr && ci->getCalledFunction() == confFunc) {
 159           confCall = ci;
 160           break;
 161         }
 162       }
 163     }
 164 
 165     // Add predecessors if call was not found
 166     pred_iterator PI = pred_begin(currentBlock), E = pred_end(currentBlock);
 167     assert(PI != E or confCall != nullptr && "Could not find kernel configuration call!");
 168     for (; PI != E; ++PI) {
 169       q.push(*PI);
 170     }
 171   }
 172 
 173   assert(confCall != nullptr && "Kernel configuration call not found!");
 174 
 175   return confCall;
 176 }

 178 void getKernelLaunchConfig(llvm::Module &m, llvm::CallBase *kernelLaunchSite, std::vector<llvm::Value *> &config) {
 179   CallBase *confCall = getKernelConfigCall(m, kernelLaunchSite);
 180 
 181   for (Use &val : confCall->operands()) {
 182     // only the arguments are wanted not the function itself
 183     if (dyn_cast_or_null<Function>(val.get()) != nullptr)
 184       continue;
 185     // discard basic blocks in case the callbase is an invoke instruction
 186     if (dyn_cast_or_null<BasicBlock>(val.get()) != nullptr)
 187       continue;
 188     config.push_back(val.get());
 189   }
 190   return;
 191 }

 268 /// Creates a wrapper for a cuda kernel in module m with the given name
 269 /// The functiontype is the the function signature of the kernel
 270 Function *createKernelWrapper(llvm::Module &m, const std::string name, FunctionType *ft) {
 271   LLVMContext &ctx = m.getContext();
 272 
 273   IRBuilder<> builder(ctx);
 274   FunctionType *launchType =
 275       FunctionType::get(builder.getInt32Ty(),
 276                         {builder.getInt8PtrTy(), builder.getInt64Ty(), builder.getInt32Ty(), builder.getInt64Ty(),
 277                          builder.getInt32Ty(), builder.getInt8PtrTy()->getPointerTo(), builder.getInt64Ty(),
 278                          m.getTypeByName("struct.CUstream_st")->getPointerTo()},
 279                         false);
 280   // alternativ:
 281   // m.getFunction("cudaConfigureCall")->Type()-><LetzterTypderArgumente>
 282 
 283   Function *launchKernel = dyn_cast<Function>(m.getOrInsertFunction("cudaLaunchKernel", launchType).getCallee());
 284 
 285   // Same convention like the cuda api function: grid, and block sizes are i32
 286   // variables and the first two are merged to one i64
 287   vector<Type *> wrapperParams = {builder.getInt64Ty(), builder.getInt32Ty(),
 288                                   builder.getInt64Ty(), builder.getInt32Ty(),
 289                                   builder.getInt64Ty(), m.getTypeByName("struct.CUstream_st")->getPointerTo()};
 290 
 291   // add kernel parameters
 292   for (auto *param : ft->params())
 293     wrapperParams.push_back(param);
 294 
 295   FunctionType *wrapperType = FunctionType::get(builder.getInt32Ty(), wrapperParams, false);
 296 
 297   Function *wrapper = dyn_cast<Function>(m.getOrInsertFunction(name, wrapperType).getCallee());
 298 
 299   BasicBlock *entry = BasicBlock::Create(ctx, "entry", wrapper, nullptr);
 300 
 301   builder.SetInsertPoint(entry);
 302 
 303   // Get the 8 Arguments for cudaLaunchKernel (dim3 is translated to i64 + i32)
 304   // cudaLaunchKernel ( const void* func, dim3 gridDim, dim3 blockDim,
 305   //                    void** args, size_t sharedMem, cudaStream_t stream )
 306   vector<Value *> launchKernelArgs;
 307 
 308   // 1. function handle
 309   Value *fuptr = builder.CreatePointerCast(wrapper, builder.getInt8PtrTy());
 310   launchKernelArgs.push_back(fuptr);
 311 
 312   // 2.-5. gridDim and blockDim
 313   auto it = wrapper->arg_begin();
 314   // launchKernelArgs.push_back(gridXY);  // Skip Original Grid XY  Value with
 315   // fixed one
 316   launchKernelArgs.push_back(&*it);
 317   ++it;
 318   // Grid Z
 319   launchKernelArgs.push_back(&*it);
 320   ++it;
 321   // launchKernelArgs.push_back(blockXY);   // Skip original block XY value with
 322   // fixed one
 323   launchKernelArgs.push_back(&*it);
 324   ++it;
 325   launchKernelArgs.push_back(&*it);
 326   ++it;
 327 
 328   Value *shm = &*(it++);    // Shared Memory
 329   Value *stream = &*(it++); // Stream
 330 
 331   // 6. pointer to argument ptr array
 332   Value *ptr_array = builder.CreateAlloca(builder.getInt8PtrTy(), builder.getInt64(ft->params().size()));
 333 
 334   for (int i = 0; i < ft->params().size(); ++i) {
 335     // Get the parameter for the kernel, which start after the grid etc
 336     Value *argument = wrapper->arg_begin() + 6 + i;
 337     Value *memptr = builder.CreateAlloca(argument->getType());
 338     Value *store = builder.CreateStore(argument, memptr);
 339     Value *strptr = builder.CreateGEP(ptr_array, builder.getInt64(i));
 340     strptr = builder.CreatePointerCast(strptr, argument->getType()->getPointerTo()->getPointerTo());
 341     builder.CreateStore(memptr, strptr);
 342   }
 343 
 344   launchKernelArgs.push_back(ptr_array);
 345 
 346   // 7. + 8.
 347   launchKernelArgs.push_back(shm);
 348   launchKernelArgs.push_back(stream);
 349 
 350   Value *ret = builder.CreateCall(launchKernel, launchKernelArgs);
 351 
 352   builder.CreateRet(ret);
 353 
 354   return wrapper;
 355 }

 373 llvm::CallBase *replaceKernelLaunch(llvm::Module &m, llvm::CallBase *kernelLaunchSite,
 374                                     llvm::Function *replacementWrapper,
 375                                     std::vector<llvm::Value *> &additionalArguments) {
 376 
 377   // gather launch config and kernel arguments
 378   std::vector<Value *> config;
 379   getKernelLaunchConfig(m, kernelLaunchSite, config);
 380   std::vector<Value *> kargs;
 381   getKernelArguments(kernelLaunchSite, kargs);
 382   CallBase *confCall = getKernelConfigCall(m, kernelLaunchSite);
 383   assert(confCall != nullptr);
 384 
 385   LLVMContext &ctx = m.getContext();
 386   IRBuilder<> builder(ctx);
 387 
 388   // check if config call is invoke inst and insert branch if so
 389   InvokeInst *confInv = dyn_cast_or_null<InvokeInst>(confCall);
 390   if (confInv != nullptr) {
 391     BasicBlock *branchTarget = confInv->getNormalDest();
 392     builder.SetInsertPoint(confCall);
 393     builder.CreateBr(branchTarget);
 394   }
 395 
 396   // replace all uses of confCall with 0 (for success) and erase it
 397   confCall->replaceAllUsesWith(builder.getInt32(0));
 398   confCall->eraseFromParent();
 399 
 400   // Prepare Launch Call Args
 401   std::vector<Value *> args;
 402   // append kernel launch config
 403   for (auto &val : config) {
 404     args.push_back(val);
 405   }
 406   // append kernel arguments
 407   for (auto &val : kargs) {
 408     args.push_back(val);
 409   }
 410   // append additional arguments
 411   for (auto &val : additionalArguments) {
 412     args.push_back(val);
 413   }
 414 
 415   // insert before old kernel launch
 416   builder.SetInsertPoint(kernelLaunchSite);
 417 
 418   FunctionType *ft = replacementWrapper->getFunctionType();
 419   auto params = ft->params();
 420   // cast pointer to stream if types do not match
 421   if (args[5]->getType() != params[5]) {
 422     args[5] = builder.CreateBitCast(args[5], params[5]);
 423   }
 424 
 425   // insert new kernel launch
 426   CallBase *newLaunchCall = builder.CreateCall(replacementWrapper, args);
 427 
 428   // check if kernelLaunch is invoke inst and insert branch if so
 429   InvokeInst *kernelInv = dyn_cast_or_null<InvokeInst>(kernelLaunchSite);
 430   if (kernelInv != nullptr) {
 431     BasicBlock *branchTarget = kernelInv->getNormalDest();
 432     builder.SetInsertPoint(kernelLaunchSite);
 433     builder.CreateBr(branchTarget);
 434   }
 435 
 436   // remove old kernel launch
 437   kernelLaunchSite->eraseFromParent();
 438 
 439   return newLaunchCall;
 440 }
