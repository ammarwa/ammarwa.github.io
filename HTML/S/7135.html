   7 /*
   8            An implementation of top-down splaying with sizes
   9              D. Sleator <sleator@cs.cmu.edu>, January 1994.
  10 
  11   This extends top-down-splay.c to maintain a size field in each node.
  12   This is the number of nodes in the subtree rooted there.  This makes
  13   it possible to efficiently compute the rank of a key.  (The rank is
  14   the number of nodes to the left of the given key.)  It it also
  15   possible to quickly find the node of a given rank.  Both of these
  16   operations are illustrated in the code below.  The remainder of this
  17   introduction is taken from top-down-splay.c.
  18 
  19   "Splay trees", or "self-adjusting search trees" are a simple and
  20   efficient data structure for storing an ordered set.  The data
  21   structure consists of a binary tree, with no additional fields.  It
  22   allows searching, insertion, deletion, deletemin, deletemax,
  23   splitting, joining, and many other operations, all with amortized
  24   logarithmic performance.  Since the trees adapt to the sequence of
  25   requests, their performance on real access patterns is typically even
  26   better.  Splay trees are described in a number of texts and papers
  27   [1,2,3,4].
  28 
  29   The code here is adapted from simple top-down splay, at the bottom of
  30   page 669 of [2].  It can be obtained via anonymous ftp from
  31   spade.pc.cs.cmu.edu in directory /usr/sleator/public.
  32 
  33   The chief modification here is that the splay operation works even if the
  34   item being splayed is not in the tree, and even if the tree root of the
  35   tree is NULL.  So the line:
  36 
  37                               t = splay(i, t);
  38 
  39   causes it to search for item with key i in the tree rooted at t.  If it's
  40   there, it is splayed to the root.  If it isn't there, then the node put
  41   at the root is the last one before NULL that would have been reached in a
  42   normal binary search for i.  (It's a neighbor of i in the tree.)  This
  43   allows many other operations to be easily implemented, as shown below.
  44 
  45   [1] "Data Structures and Their Algorithms", Lewis and Denenberg,
  46        Harper Collins, 1991, pp 243-251.
  47   [2] "Self-adjusting Binary Search Trees" Sleator and Tarjan,
  48        JACM Volume 32, No 3, July 1985, pp 652-686.
  49   [3] "Data Structure and Algorithm Analysis", Mark Weiss,
  50        Benjamin Cummins, 1992, pp 119-130.
  51   [4] "Data Structures, Algorithms, and Performance", Derick Wood,
  52        Addison-Wesley, 1993, pp 367-375
  53 */

  55 Tree * splay (T i, Tree *t)
  56 /* Splay using the key i (which may or may not be in the tree.) */
  57 /* The starting root is t, and the tree used is defined by rat  */
  58 /* size fields are maintained */
  59 {
  60     Tree N, *l, *r, *y;
  61     T comp,l_size, r_size;
  62     if (t == NULL) return t;
  63     N.left = N.right = NULL;
  64     l = r = &N;
  65     l_size = r_size = 0;
  66  
  67     for (;;) {
  68         comp = compare(i, t->key);
  69         if (comp < 0) {
  70             if (t->left == NULL) break;
  71             if (compare(i, t->left->key) < 0) {
  72                 y = t->left;                           /* rotate right */
  73                 t->left = y->right;
  74                 y->right = t;
  75                 t->size = node_size(t->left) + node_size(t->right) + 1;
  76                 t = y;
  77                 if (t->left == NULL) break;
  78             }
  79             r->left = t;                               /* link right */
  80             r = t;
  81             t = t->left;
  82             r_size += 1+node_size(r->right);
  83         } else if (comp > 0) {
  84             if (t->right == NULL) break;
  85             if (compare(i, t->right->key) > 0) {
  86                 y = t->right;                          /* rotate left */
  87                 t->right = y->left;
  88                 y->left = t;
  89                 t->size = node_size(t->left) + node_size(t->right) + 1;
  90                 t = y;
  91                 if (t->right == NULL) break;
  92             }
  93             l->right = t;                              /* link left */
  94             l = t;
  95             t = t->right;
  96             l_size += 1+node_size(l->left);
  97         } else {
  98             break;
  99         }
 100     }
 101     l_size += node_size(t->left);  /* Now l_size and r_size are the sizes of */
 102     r_size += node_size(t->right); /* the left and right trees we just built.*/
 103     t->size = l_size + r_size + 1;
 104 
 105     l->right = r->left = NULL;
 106 
 107     /* The following two loops correct the size fields of the right path  */
 108     /* from the left child of the root and the right path from the left   */
 109     /* child of the root.                                                 */
 110     for (y = N.right; y != NULL; y = y->right) {
 111         y->size = l_size;
 112         l_size -= 1+node_size(y->left);
 113     }
 114     for (y = N.left; y != NULL; y = y->left) {
 115         y->size = r_size;
 116         r_size -= 1+node_size(y->right);
 117     }
 118  
 119     l->right = t->left;                                /* assemble */
 120     r->left = t->right;
 121     t->left = N.right;
 122     t->right = N.left;
 123 
 124     return t;
 125 }

 127 Tree * insert(T i, Tree * t) {
 128 /* Insert key i into the tree t, if it is not already there. */
 129 /* Return a pointer to the resulting tree.                   */
 130     Tree * new;
 131 
 132     if (t != NULL) {
 133         t = splay(i,t);
 134         if (compare(i, t->key)==0) {
 135             return t;  /* it's already there */
 136         }
 137     }
 138     new = (Tree *) malloc (sizeof (Tree));
 139     if (new == NULL) {printf("Ran out of space\n"); exit(1);}
 140     if (t == NULL) {
 141         new->left = new->right = NULL;
 142     } else if (compare(i, t->key) < 0) {
 143         new->left = t->left;
 144         new->right = t;
 145         t->left = NULL;
 146         t->size = 1+node_size(t->right);
 147     } else {
 148         new->right = t->right;
 149         new->left = t;
 150         t->right = NULL;
 151         t->size = 1+node_size(t->left);
 152     }
 153     new->key = i;
 154     new->size = 1 + node_size(new->left) + node_size(new->right);
 155     return new;
 156 }

 158 Tree * delete(T i, Tree *t) {
 159 /* Deletes i from the tree if it's there.               */
 160 /* Return a pointer to the resulting tree.              */
 161     Tree * x;
 162     T tsize;
 163 
 164     if (t==NULL) return NULL;
 165     tsize = t->size;
 166     t = splay(i,t);
 167     if (compare(i, t->key) == 0) {               /* found it */
 168         if (t->left == NULL) {
 169             x = t->right;
 170         } else {
 171             x = splay(i, t->left);
 172             x->right = t->right;
 173         }
 174         free(t);
 175         if (x != NULL) {
 176             x->size = tsize-1;
 177         }
 178         return x;
 179     } else {
 180         return t;                         /* It wasn't there */
 181     }
 182 }

 184 Tree *find_rank(T r, Tree *t) {
 185 /* Returns a pointer to the node in the tree with the given rank.  */
 186 /* Returns NULL if there is no such node.                          */
 187 /* Does not change the tree.  To guarantee logarithmic behavior,   */
 188 /* the node found here should be splayed to the root.              */
 189     T lsize;
 190     if ((r < 0) || (r >= node_size(t))) return NULL;
 191     for (;;) {
 192         lsize = node_size(t->left);
 193         if (r < lsize) {
 194             t = t->left;
 195         } else if (r > lsize) {
 196             r = r - lsize -1;
 197             t = t->right;
 198         } else {
 199             return t;
 200         }
 201     }
 202 }
