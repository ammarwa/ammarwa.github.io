  38 #define ULL unsigned long long int
  39 #define USE_ASYNC_STREAM

  41 class ChannelDev {
  42   private:
  43     int id;
  44     volatile int *doorbell;
  45 
  46     uint8_t *buff;
  47     uint8_t *buff_end;
  48 
  49     /* head/tail pointers */
  50     uint8_t *volatile buff_write_head_ptr;
  51     uint8_t *volatile buff_write_tail_ptr;
  52 
  53   public:
  54     ChannelDev() {}
  55 
  56     __device__ __forceinline__ void push(void *packet, uint32_t nbytes) {
  57         assert(nbytes != 0);
  58 
  59         uint8_t *curr_ptr = NULL;
  60 
  61         while (curr_ptr == NULL) {
  62             curr_ptr =
  63                 (uint8_t *)atomicAdd((ULL *)&buff_write_head_ptr, (ULL)nbytes);
  64 
  65             /* if the current position plus nbytes is after buff_end, the
  66              * buffer is full.
  67              * Many warps could find condition true, but only the first warp
  68              * will find true the condition after. */
  69             if (curr_ptr + nbytes > buff_end) {
  70                 /* I am the first warp that found the buffer full and
  71                  * I am the one responsible for flushing the buffer out */
  72                 if (curr_ptr <= buff_end) {
  73                     /* wait until everyone completed to write */
  74                     while (buff_write_tail_ptr != curr_ptr) {
  75                     }
  76 
  77                     /* flush buffer */
  78                     flush();
  79                 } else {
  80                     /* waiting for buffer to flush */
  81                     while (buff_write_head_ptr > buff_end) {
  82                     }
  83                 }
  84                 curr_ptr = NULL;
  85             }
  86         }
  87 
  88         memcpy(curr_ptr, packet, nbytes);
  89         atomicAdd((ULL *)&buff_write_tail_ptr, (ULL)nbytes);
  90     }
  91 
  92     __device__ __forceinline__ void flush() {
  93         uint32_t nbytes = (uint32_t)(buff_write_tail_ptr - buff);
  94         // printf("FLUSH CHANNEL#%d: buffer bytes %d\n", id, nbytes);
  95         if (nbytes == 0) {
  96             return;
  97         }
  98 
  99         /* make sure everything is visible in memory */
 100         __threadfence_system();
 101 
 102         assert(*doorbell == 0);
 103         /* notify current buffer has something*/
 104         *doorbell = nbytes;
 105         __threadfence_system();
 106 
 107         /* wait for host to release the doorbell */
 108         while (*doorbell != 0)
 109             ;
 110 
 111         /* reset head/tail */
 112         buff_write_tail_ptr = buff;
 113         __threadfence();
 114         buff_write_head_ptr = buff;
 115 
 116         //  printf("FLUSH CHANNEL#%d: DONE\n", id);
 117     }
 118 
 119   private:
 120     /* called by the ChannelHost init */
 121     void init(int id, int *h_doorbell, int buff_size) {
 122         CUDA_SAFECALL(cudaHostGetDevicePointer((void **)&doorbell,
 123                                                (void *)h_doorbell, 0));
 124 
 125         /* allocate large buffer */
 126 #ifdef USE_ASYNC_STREAM
 127         CUDA_SAFECALL(cudaMalloc((void **)&buff, buff_size));
 128 #else
 129         CUDA_SAFECALL(cudaMallocManaged((void **)&buff, buff_size));
 130 #endif
 131         buff_write_head_ptr = buff;
 132         buff_write_tail_ptr = buff;
 133         buff_end = buff + buff_size;
 134         this->id = id;
 135     }
 136 
 137     friend class ChannelHost;
 138 };

 140 class ChannelHost {
 141   private:
 142     volatile int *doorbell;
 143 
 144     cudaStream_t stream;
 145     ChannelDev *ch_dev;
 146 
 147     /* pointers to device buffer */
 148     uint8_t *dev_buff_read_head;
 149     uint8_t *dev_buff;
 150 
 151     /* receiving thread */
 152     pthread_t thread;
 153     volatile bool thread_started;
 154 
 155   public:
 156     int id;
 157     int buff_size;
 158 
 159   public:
 160     ChannelHost() {}
 161 
 162     void init(int id, int buff_size, ChannelDev *ch_dev,
 163               void *(*thread_fun)(ChannelHost *)) {
 164         this->buff_size = buff_size;
 165         this->id = id;
 166         /* get device properties */
 167         cudaDeviceProp prop;
 168         int device = 0;
 169         cudaGetDeviceProperties(&prop, device);
 170         if (prop.canMapHostMemory == 0) {
 171             CUDA_SAFECALL(cudaSetDeviceFlags(cudaDeviceMapHost));
 172         }
 173 
 174 #ifdef USE_ASYNC_STREAM
 175         /* create stream that will read memory with highest possible priority */
 176         int priority_high, priority_low;
 177         CUDA_SAFECALL(
 178             cudaDeviceGetStreamPriorityRange(&priority_low, &priority_high));
 179         CUDA_SAFECALL(cudaStreamCreateWithPriority(
 180             &stream, cudaStreamNonBlocking, priority_high));
 181 #endif
 182 
 183         /* create doorbell */
 184         CUDA_SAFECALL(cudaHostAlloc((void **)&doorbell, sizeof(int),
 185                                     cudaHostAllocMapped));
 186         /* set doorbell to zero */
 187         *doorbell = 0;
 188 
 189         /* initialize device channel */
 190         this->ch_dev = ch_dev;
 191         ch_dev->init(id, (int *)doorbell, buff_size);
 192 
 193         dev_buff = ch_dev->buff;
 194         dev_buff_read_head = dev_buff;
 195         if (thread_fun != NULL) {
 196             thread_started = true;
 197             pthread_create(&thread, NULL, (void *(*)(void *))thread_fun,
 198                            (void *)this);
 199         } else {
 200             thread_started = false;
 201         }
 202     }
 203 
 204     /* when used in nvbit we don't want to dealloc because
 205      * when modules are unloaded the driver automatically
 206      * deallocates CUDA malloc, so further deallocs done
 207      * here will result in errors */
 208     void destroy(bool dealloc) {
 209         if (thread_started) {
 210             thread_started = false;
 211             pthread_join(thread, NULL);
 212         }
 213         if (dealloc) {
 214 #ifdef USE_ASYNC_STREAM
 215             CUDA_SAFECALL(cudaStreamDestroy(stream));
 216 #endif
 217             CUDA_SAFECALL(cudaFree((int *)doorbell));
 218             CUDA_SAFECALL(cudaFree(ch_dev->buff));
 219         }
 220     }
 221 
 222     bool is_active() { return thread_started; }
 223 
 224     uint32_t recv(void *buff, uint32_t max_buff_size) {
 225         assert(max_buff_size > 0);
 226         assert(doorbell != NULL);
 227         uint32_t buff_nbytes = *doorbell;
 228         if (buff_nbytes == 0) {
 229             return 0;
 230         }
 231         int nbytes = buff_nbytes;
 232 
 233         if (buff_nbytes > max_buff_size) {
 234             nbytes = max_buff_size;
 235         }
 236 #ifdef USE_ASYNC_STREAM
 237         CUDA_SAFECALL(cudaMemcpyAsync(buff, dev_buff_read_head, nbytes,
 238                                       cudaMemcpyDeviceToHost, stream));
 239         CUDA_SAFECALL(cudaStreamSynchronize(stream));
 240 #else
 241         memcpy(buff, dev_buff_read_head, nbytes);
 242 #endif
 243         int bytes_left = buff_nbytes - nbytes;
 244         assert(bytes_left >= 0);
 245         if (bytes_left > 0) {
 246             dev_buff_read_head += nbytes;
 247         } else {
 248             dev_buff_read_head = dev_buff;
 249         }
 250 
 251         *doorbell = bytes_left;
 252         // printf("HOST RECEIVED nbytes %d - bytes left %d\n", nbytes,
 253         // bytes_left);
 254         return nbytes;
 255     }
 256 
 257     pthread_t get_thread() { return thread; }
 258 
 259     friend class MultiChannelHost;
 260 };
