  18 std::string getModulePrefix(llvm::Module *m) {
  19   const std::string sourceFile = m->getSourceFileName();
  20 
  21   // Get BaseName
  22   std::string baseName = "";
  23   size_t pos = sourceFile.rfind("/");
  24   if (pos != std::string::npos)
  25     baseName = sourceFile.substr(pos + 1);
  26   else
  27     baseName = sourceFile;
  28 
  29   std::ifstream file(sourceFile, std::ios::binary | std::ios::ate);
  30   auto size = file.tellg();
  31   file.seekg(0, std::ios::beg);
  32 
  33   std::string buffer(size, '\0');
  34   assert(file.read(&buffer[0], size && "Could not read file!"));
  35 
  36   auto hash = generateHash(buffer.c_str(), size);
  37 
  38   std::ostringstream oss;
  39   oss << baseName << "_" << std::hex << hash;
  40   return std::string(oss.str());
  41 }

  43 std::map<llvm::BasicBlock *, int>
  44 getBlockIDMap(llvm::Function *func, std::vector<mekong::PTXFunction> funcVec,
  45               std::string originalFunctionName) {
  46   PTXFunction funcPTX;
  47   for (int i = 0; i < funcVec.size(); ++i) {
  48     if (originalFunctionName == funcVec[i].name)
  49       funcPTX = funcVec[i];
  50   }
  51 
  52   // Build BB Name -> ID Map
  53   std::map<StringRef, int> nameMap;
  54   for (int i = 0; i < funcPTX.bb.size(); ++i) {
  55     nameMap.insert({funcPTX.bb[i].name, i});
  56   }
  57 
  58   // Build Final Map
  59   std::map<llvm::BasicBlock *, int> ptrMap;
  60   for (llvm::BasicBlock &bb : func->getBasicBlockList()) {
  61     ptrMap[&bb] = nameMap[bb.getName()];
  62   }
  63 
  64   return ptrMap;
  65 }

  67 void dumpModuleToFile(llvm::Module &m, std::string filepath) {
  68 
  69   std::error_code EC;
  70   { // Seperate name space to prevent usage of file later in code
  71     // also flushes input of file to disk - very important!
  72     raw_fd_ostream file(filepath, EC, llvm::sys::fs::OpenFlags(1)); // OF_Text
  73     // WriteBitcodeToFile()
  74     assert(!EC);
  75     llvm::WriteBitcodeToFile(m, file);
  76     file.flush();
  77   }
  78 }

  92 llvm::GlobalVariable *createGlobalString(llvm::Module &M, llvm::StringRef Str,
  93                                          const llvm::Twine &Name,
  94                                          unsigned AddressSpace) {
  95   Constant *StrConstant = ConstantDataArray::getString(M.getContext(), Str);
  96   auto *GV = new GlobalVariable(
  97       M, StrConstant->getType(), true, GlobalValue::PrivateLinkage, StrConstant,
  98       Name, nullptr, GlobalVariable::NotThreadLocal, AddressSpace);
  99   GV->setUnnamedAddr(GlobalValue::UnnamedAddr::Global);
 100   // GV->setAlignment(1);
 101   GV->setAlignment(MaybeAlign(1));
 102 
 103   return GV;
 104 }

 117 void linkIR(llvm::StringRef ir, llvm::Module &m) {
 118   llvm::LLVMContext &ctx = m.getContext();
 119   llvm::Linker linker(m);
 120   llvm::SMDiagnostic err;
 121   std::unique_ptr<llvm::Module> irModule = llvm::parseIR(
 122       llvm::MemoryBufferRef(*llvm::MemoryBuffer::getMemBuffer(ir, "", true)),
 123       err, ctx);
 124 
 125   if (!irModule)
 126     err.print("linkIR", llvm::errs());
 127 
 128   linker.linkInModule(std::move(irModule));
 129 }

 144 llvm::CallInst *callPrintf(llvm::Module &module, std::string str,
 145                            llvm::Value *val) {
 146   llvm::LLVMContext &ctx = module.getContext();
 147   llvm::ArrayType *type =
 148       llvm::ArrayType::get(llvm::IntegerType::get(ctx, 8), str.size() + 1);
 149   llvm::Constant *strConstant = llvm::ConstantDataArray::getString(ctx, str);
 150   llvm::GlobalVariable *GVStr = new llvm::GlobalVariable(
 151       module, type, true, llvm::GlobalValue::InternalLinkage, strConstant);
 152   llvm::Constant *zero =
 153       llvm::Constant::getNullValue(llvm::IntegerType::getInt32Ty(ctx));
 154 
 155   llvm::IRBuilder<> builder(module.getContext());
 156   llvm::Value *strVal = builder.CreateGEP(GVStr, {zero, zero});
 157 
 158   std::vector<llvm::Value *> args;
 159   args.push_back(&*strVal);
 160   if (val != nullptr)
 161     args.push_back(val);
 162 
 163   llvm::Function *printf_func = getPrintfFunc(module);
 164   printf_func->setCallingConv(llvm::CallingConv::C);
 165   llvm::CallInst *ci = builder.CreateCall(printf_func, args);
 166   return ci;
 167 }
