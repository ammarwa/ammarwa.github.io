  57 void parda_process(char* input, T tim, program_data_t* pdt) {
  58         GHashTable *gh = pdt->gh;
  59         Tree* root = pdt->root;
  60         narray_t* ga = pdt->ga;
  61         unsigned int *histogram = pdt->histogram;
  62         int distance;
  63         T *lookup;
  64         lookup = g_hash_table_lookup(gh, input);
  65         if (lookup == NULL) {
  66                 char* data = strdup(input);
  67                 root = insert(tim, root);
  68                 T *p_data;
  69 
  70                 narray_append_val(ga, input);
  71                 if (!(p_data = (T*)malloc(sizeof(T)))) exit(1);
  72                 *p_data = tim;
  73                 g_hash_table_insert(gh, data, p_data);  // Store pointer to list element
  74         }
  75 
  76         // Hit: We've seen this data before
  77         else {
  78                 root = insert((*lookup), root);
  79                 distance = node_size(root->right);
  80                 root = delete(*lookup, root);
  81                 root = insert(tim, root);
  82                 int *p_data;
  83                 if (!(p_data = (int*)malloc(sizeof(int)))) exit(1);
  84                 *p_data = tim;
  85                 g_hash_table_replace(gh, strdup(input), p_data);
  86 
  87                 // Is distance greater than the largest bucket
  88                 if (distance > nbuckets)
  89                         histogram[B_OVFL]++;
  90                 else
  91                         histogram[distance]++;
  92         }
  93         pdt->root = root;
  94 }

 107 void parda_get_abfront(program_data_t* pdt_a, const narray_t* gb, const processor_info_t* pit_a) {
 108         //printf("enter abfront and before for loopheihei\n");
 109         T tim = pit_a->tend + 1;
 110         GHashTable* gh = pdt_a->gh;
 111         narray_t* ga = pdt_a->ga;
 112         Tree* root = pdt_a->root;
 113         unsigned* histogram = pdt_a->histogram;
 114         int i;
 115         T *lookup;
 116         T distance;
 117         unsigned len = narray_get_len(gb);
 118         //printf("enter abfront and before for loop\n");
 119         for(i=0; i < len; i++, tim++) {
 120                 HKEY entry;
 121                 char* temp=((HKEY*)gb->data)[i];
 122                 strcpy(entry, temp);
 123                 lookup = g_hash_table_lookup(gh, entry);
 124                 //printf("merge entry %s\n",entry);
 125                 if(lookup==NULL) {
 126                         narray_append_val (ga, entry);
 127                         root = insert(tim, root);
 128                 } else {
 129                         root = insert((*lookup), root);
 130                         distance = node_size(root->right);
 131                         root = delete(*lookup, root);
 132                         root = insert(tim, root);
 133                         if (distance > nbuckets)
 134                                 histogram[B_OVFL]++;
 135                         else
 136                                 histogram[distance]++;
 137                 }
 138         }
 139         pdt_a->root = root;
 140         pdt_a->gh = gh;
 141 }

 143 int parda_get_abend(program_data_t* pdt_b,
 144         const   end_keytime_t* ekt_a ) {
 145         Tree* root = pdt_b->root;
 146         GHashTable* gh = pdt_b->gh;
 147         narray_t* gkeys = ekt_a->gkeys;
 148         narray_t* gtimes = ekt_a->gtimes;
 149         unsigned len = narray_get_len(gkeys);
 150         unsigned i;
 151         HKEY key;
 152         T tim;
 153         T* lookup;
 154         for (i = 0; i < len; i++) {
 155                 char* temp = ((HKEY*)gkeys->data)[i];
 156                 strcpy(key, temp);
 157                 tim = ((T*)(gtimes->data))[i];
 158                 lookup = g_hash_table_lookup(gh, key);
 159                 if (lookup == NULL) {
 160                         char* data = strdup(key);
 161                         root = insert(tim,root);
 162                         T *p_data;
 163                         if ( !(p_data = (T*)malloc(sizeof(T))) ) return -1;
 164                         *p_data = tim;
 165                         g_hash_table_insert(gh, data, p_data);
 166                 }
 167         }
 168         pdt_b->root = root;
 169         pdt_b->gh = gh;
 170 
 171         return 0;

 174 program_data_t parda_merge(program_data_t* pdt_a, program_data_t* pdt_b,
 175         const processor_info_t* pit_b) {
 176         program_data_t pdt;
 177         parda_get_abfront(pdt_a, pdt_b->ga, pit_b);
 178         DEBUG(printf("after get abfront %d\n", pit_b->pid);)
 179         narray_free(pdt_b->ga);
 180         pdt_a->ekt = parda_generate_end(pdt_a);
 181         parda_get_abend(pdt_b, &pdt_a->ekt);
 182         narray_free(pdt_a->ekt.gkeys);
 183         narray_free(pdt_a->ekt.gtimes);
 184         pdt.ga = pdt_a->ga;
 185         pdt.root = pdt_b->root;
 186         pdt.gh = pdt_b->gh;
 187         pdt.histogram = pdt_a->histogram;
 188         int i;
 189         for (i = 0; i < nbuckets+2; i++)
 190                 pdt.histogram[i] += pdt_b->histogram[i];
 191         free(pdt_b->histogram);
 192 
 193         return pdt;
 194 }
